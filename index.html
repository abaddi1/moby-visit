// ==UserScript== // @name Auto Click Specific Links // @namespace http://tampermonkey.net/ // @version 1.0 // @description Auto click on specific links when the page loads // @match https://www.mobyscreener.com/* // @grant none // ==/UserScript== (function() { 'use strict'; // List of target links const targetLinks = [ "https://www.mobyscreener.com/solana/2B5Rs88FYyFp4XciFrJ7zovvEeybiinZA8PXsSptBAGS", "https://www.mobyscreener.com/solana/mkvXiNBpa8uiSApe5BrhWVJaT87pJFTZxRy7zFapump" ]; window.addEventListener('load', () => { setTimeout(() => { const links = document.querySelectorAll('a'); for (let link of links) { if (targetLinks.includes(link.href)) { link.click(); break; } } }, 5000); // wait 5 seconds before clicking }); })();

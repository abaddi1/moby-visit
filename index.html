// سكريبت أوتوماتيكي كامل window.addEventListener('load', function () { // مهلة قصيرة باش الصفحة تكمل تحميلها setTimeout(() => { // سكرول تدريجي لأسفل الصفحة let scrollInterval = setInterval(() => { window.scrollBy(0, 50); if ((window.innerHeight + window.scrollY) >= document.body.offsetHeight) { clearInterval(scrollInterval); // بعد ما يوصل للأسفل، يختار عملة عشوائية setTimeout(() => { let coins = document.querySelectorAll('a, button, div'); coins = Array.from(coins).filter(el => el.innerText && el.innerText.match(/\b[A-Z]{2,5}\b/)); // نصوص مختصرة مثل BTC, ETH if (coins.length > 0) { let randomCoin = coins[Math.floor(Math.random() * coins.length)]; randomCoin.click(); console.log('✅ تم النقر على عملة:', randomCoin.innerText); } else { console.log('❌ لم يتم العثور على عملات'); } }, 1000); } }, 100); }, 2000); });
